% ============================================================
% Top header
% ============================================================

\NewDocumentCommand{\Header}{m m m m}{%
    \begingroup
    \centering
        {\NameStyle #1\par}%
        \vspace{\SmallSep}%
        {\ContactStyle
            #2%
            \ $\mid$\  \href{tel:#3}{#3}%
            \ $\mid$\  \href{mailto:#4}{\uline{#4}}%
            \par
        }%
    \endgroup
    \vspace{\LargeSep}%
}

% Usage:
% \Header
%   {Julien Berman}
%   {Chevy Chase, MD}
%   {(240) 454-2273}
%   {julien.d.berman@gmail.com}

% ============================================================
% Section header
% ============================================================

\NewDocumentCommand{\SectionHeader}{m}{
    \noindent
    \begingroup
    \setlength{\fboxsep}{4pt}
    \colorbox{black}{
        \makebox[\dimexpr\textwidth-2\fboxsep\relax][c]{
            \color{white}\SectionHeaderStyle #1
        }
    }
    \endgroup
    \par\vspace{\MedSep}
}

% Usage:
% \SectionHeader
%     {Education}

% ============================================================
% Education entry
% ============================================================
\NewDocumentCommand{\EducationEntry}{m m m o}{%
    \Needspace{3\baselineskip}%
    \begin{tabularx}{\textwidth}{@{}L D@{}}
        {\OrgStyle #1}{\LocationStyle\ ---\ #2} & {\DateStyle #3}%
        \IfValueT{#4}{%
            \\[\SmallSep]
            \multicolumn{2}{@{}p{\textwidth}@{}}{%
                \hspace*{\Indent}%
                \parbox[t]{\dimexpr\textwidth-\Indent\relax}{%
                    \BodyStyle #4%
                }%
            }%
        }%
    \end{tabularx}%
    \par\vspace{\MedSep}%
}

% Usage:
% \EducationEntry
%     {Harvard University}
%     {Cambridge, MA}
%     {May 2026}
%     [Bachelor of Arts.]

% ============================================================
% Generic entry
% ============================================================

\NewDocumentCommand{\Entry}{m o +m}{%
    \Needspace{4\baselineskip}%
    {\OrgStyle #1}%
    \IfValueT{#2}{ {\LocationStyle --- #2}}%
    \par\vspace{\SmallSep}%
    #3%
    \par\vspace{\MedSep}%
}


% ============================================================
% Position
% ============================================================

\NewDocumentCommand{\Position}{m m +m}{%
    \Needspace{3\baselineskip}%
    \begin{tabularx}{\textwidth}{@{}I D@{}}
        {\RoleStyle #1} & {\DateStyle #2}
    \end{tabularx}%
    \begin{Bullets}
        #3
    \end{Bullets}%
}
